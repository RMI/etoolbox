def setup_logging(
    filename: Path,
    config_file: Path | None = None,
    config: dict | None = None,
) -> None:
    """Set up logging with a queue handler.

    If neither config_file nor config is provided, use default logging config
    :const:`DEFAULT_LOGGING_CONFIG`.

    Args:
        filename: path to log file
        config_file: path to config file
        config: config dictionary

    """
    if config_file is not None:
        with open(config_file) as f_in:
            config = json.load(f_in)
    elif config is None:
        config = DEFAULT_LOGGING_CONFIG

    if not filename.parent.exists():
        filename.parent.mkdir(parents=True)
    config["handlers"]["file_json"]["filename"] = str(filename)

    logging.config.dictConfig(config)
    queue_handler = logging.getHandlerByName("queue_handler")
    if queue_handler is not None:
        queue_handler.listener.start()
        atexit.register(queue_handler.listener.stop)

class SafeFormatter(logging.Formatter):
    """A formatter that fills in missing extras with defaults."""

    def __init__(  # noqa: D107
        self,
        fmt=None,
        datefmt=None,
        style="%",
        validate=True,  # noqa: FBT002
        *,
        defaults=None,
        extra_defaults: dict[str, str] | None = None,
    ):
        super().__init__(fmt, datefmt, style, validate, defaults=defaults)
        self.extra_defaults = extra_defaults if extra_defaults is not None else {}

    def format(self, record: logging.LogRecord) -> str:  # noqa: D102
        for k, v in self.extra_defaults.items():
            if not hasattr(record, k):
                setattr(record, k, v)
        return super().format(record)

class JSONFormatter(logging.Formatter):
    """A JSON formatter for use in :class:`logging.Logger` objects."""

    def __init__(self, *, fmt_keys: dict[str, str] | None = None):  # noqa: D107
        super().__init__()
        self.fmt_keys = fmt_keys if fmt_keys is not None else {}



[docs]
    def format(self, record: logging.LogRecord) -> str:  # noqa: D102
        message = self._prepare_log_dict(record)
        return json.dumps(message, default=str)



    def _prepare_log_dict(self, record: logging.LogRecord):
        always_fields = {
            "message": record.getMessage(),
            "timestamp": dt.datetime.fromtimestamp(
                record.created, tz=dt.timezone.utc
            ).isoformat(),
        }
        if record.exc_info is not None:
            always_fields["exc_info"] = self.formatException(record.exc_info)

        if record.stack_info is not None:
            always_fields["stack_info"] = self.formatStack(record.stack_info)

        message = {
            key: msg_val
            if (msg_val := always_fields.pop(val, None)) is not None
            else getattr(record, val, "")
            for key, val in self.fmt_keys.items()
        }
        message.update(always_fields)

        for key, val in record.__dict__.items():
            if key not in LOG_RECORD_BUILTIN_ATTRS:
                message[key] = val

        return message
